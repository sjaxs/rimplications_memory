    \KwIn{

    }
    \KwOut{
        $The ~ lectically ~ next ~ closed ~ set, ~ after ~ A ~ if ~ it ~ exists; ~ \perp , ~ otherwise.$
    }
    \Begin{

        \ForAll{$ m \in M ~ in ~ reverse ~ order $ }{
            \If{ $ m \in A $ } {
                \ $ A = A \backslash \{m\} $\;
            } \Else {
                \ $ B = (A \cup \{m\} )'' $\;
                 \If{ $ B \backslash A ~ contains ~ no ~ element ~ < m $ }{
                     \Return $ B $ 
                 }
            }
        }
            
        \Return $ \perp $
